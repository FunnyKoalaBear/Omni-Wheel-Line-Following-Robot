#include "mbed.h"

PwmOut pwm1(D12);
PwmOut pwm2(D11);
PwmOut pwm3(PA_11_ALT0);  
PwmOut pwm4(D9);

AnalogIn S0(A0); //left side 
AnalogIn S1(A1);
AnalogIn S2(A2);
AnalogIn S3(A3);
AnalogIn S4(A4); //right side 

//const float threshold = 0.5f;
const float fast= 0.5f;
const float slow=0.1f;
const float stop=0.0f;

void drive(float left, float right) {
    pwm1.write(left);
    pwm4.write(right);
}


//variables for EOT check 
using namespace std::chrono;
Timer t;
int state = false; 
bool previously_on_line = false;



int main() {
    pwm1.period(0.01f);
    pwm4.period(0.01f);    

    while (true) {
        float right = S0.read(); //right side -2
        float rightmid = S1.read(); //-1
        float middle = S2.read(); //0
        float leftmid = S3.read(); //1
        float left = S4.read(); //left side 2
        

        bool currently_on_line = (right > 0.6 && left > 0.6);

        //end of track EOT check 

        if (currently_on_line && !previously_on_line) {
            if (state == false) {
                //first perpendiculara line passed, start timer.
                t.reset();
                t.start();
                state = true;
                printf("Right aand left positive detected!\n");
            }
            else {
                //second perpendicular line passed, compare timer. 
                t.stop();
                long long elapsed_time_ms = duration_cast<milliseconds>(t.elapsed_time()).count();
                printf("State is true!\n");
                printf("The time taken was %llu milliseconds\n", elapsed_time_ms);


                if (elapsed_time_ms > 52 && elapsed_time_ms < 10000) {
                    printf("Second signal, stopping car.\n");
                    printf("The time taken was %llu milliseconds\n", elapsed_time_ms);
                    drive(stop, stop);
                    ThisThread::sleep_for(5s);
                    state = false; 
                    printf("Strating loop again!\n");

                }
                else if (elapsed_time_ms > 10000) {
                    //took too long
                    state = false; 
                    printf("State set to false!\n"); 
                }
                else {
                    state = false; 

                }
            } 
        }

        previously_on_line = currently_on_line;
        
        
        if (middle > 0.6) {
                drive(fast, fast);
                //printf("Fast, Fast\n");
            }
        else if (leftmid > 0.6) {
            drive(slow, fast);
            //printf("Slow, Fast\n");
        }
        else if (rightmid > 0.6) {
            drive(fast, slow);
            //printf("Fast, Slow\n");
        }
        else if (left > 0.6) {
            drive(stop, fast);
            //printf("Stop, Fast\n");
        }
        else if (right > 0.6) {
            drive(fast, stop);
            //printf("Fast, Stop\n");
        }
        else {
            drive(stop, stop);
            //printf("Stop, Stop\n");
        }

        //printf("S0, Right: %.5f S1, Rightmid: %.5f S2, Middle: %.5f S3, Leftmid: %.5f S4, Left: %.5f \r\n",
        //S0.read(), S1.read(), S2.read(), S3.read(), S4.read());
        thread_sleep_for(10);
    }
    thread_sleep_for(10);

}
